\long\def$\backslash$@makecaption\#1\#2\{\%
	\vskip\abovecaptionskip{}
	\ifthenelse\{=\{table\}\{$\backslash$@currenvir\} \OR =\{listing\}\{$\backslash$@currenvir\}\}\{
		\gdef$\backslash$@captionvalue\{\#1: \#2\}\}\{
		\sbox$\backslash$@tempboxa\{\#1: \#2 \}\%
		\ifdim \wd$\backslash$@tempboxa >\hsize
			\#1: \#2 \par
		\else
			\global $\backslash$@minipagefalse
			\hb@xt@\hsize{\hfil\box\@tempboxa\hfil}\%
		\fi
		\vskip\belowcaptionskip
	\}
